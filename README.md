# PlayMusic
PlayMusic view


install: Cocoapods

  pod 'PlayMusic'

- setup image for button:

+ playButtonImage
+ pauseButtonImage
+ previousButtonImage
+ nextButtonImage
+ repeatButtonImage
example: music.repeatButtonImage(image: uiimage(name: "pause"))

- setup name of song....
+ ishiddenTitle = true
example: music.ishiddenTitle = true
         music.songName = "Huy Spring"
